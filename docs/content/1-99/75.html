75. 颜色分类

1 原题链接

颜色分类

2 解

  • 时间复杂度：O(n)
  • 空间复杂度：O(1)
class Solution {
    public void sortColors(int[] nums) {
        int num0 = 0;
        int num1 = 0;
        int num2 = 0;
        // 第一遍遍历记录下每种数字的个数
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                num0++;
            } else if (nums[i] == 1) {
                num1++;
            } else {
                num2++;
            }
        }
        // 按照题目要求的顺序来填充数字
        for (int i = 0; i < nums.length; i++) {
            if (num0 != 0) {
                nums[i] = 0;
                num0--;
            } else if (num1 != 0) {
                nums[i] = 1;
                num1--;
            } else if (num2 != 0) {
                nums[i] = 2;
                num2--;
            } else {
                break;
            }
        }
    }
}
